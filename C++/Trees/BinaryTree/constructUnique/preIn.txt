TreeNode *build(vector<int> &pre, int preStart, int preEnd, vector<int> &in, int inStart, int inEnd, map<int, int> &inMap)
{

    if (preStart > preEnd || inStart > inEnd)
        return NULL;
    TreeNode *root = new TreeNode(pre[preStart]);
    int inRoot = inMap[root->val];
    int numsLeft = inRoot - inStart;
    root->left = build(pre, preStart + 1, preStart + numsLeft, in, inStart, inRoot - 1, inMap);
    root->right = build(pre, preStart + 1 + numsLeft, preEnd, in, inRoot + 1, inEnd, inMap);
    return root;
}
TreeNode *buildTree(vector<int> &pre, vector<int> &in)
{
    map<int, int> inMap;
    for (int i = 0; i < in.size(); i++)
    {
        inMap[in[i]] = i;
    }
    TreeNode *root = build(pre, 0, pre.size() - 1, in, 0, in.size() - 1, inMap);
    return root;
}
