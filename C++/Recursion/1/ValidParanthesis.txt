// All valid paranthesisations of length n
#include <bits/stdc++.h>
using namespace std;
  
    void validParen(int n, int len, vector<string> &res, string &str, int countO, int countC)
    {
        if (len == n)
        {
            res.push_back(str);
            return;
        }
        if (countO < n / 2)
        {
            str.push_back('(');
            validParen(n, len + 1, res, str, countO + 1, countC);
            str.pop_back();
        }
        if (countC < countO)
        {
            str.push_back(')');
            validParen(n, len + 1, res, str, countO, countC + 1);
            str.pop_back();
        }
    }
