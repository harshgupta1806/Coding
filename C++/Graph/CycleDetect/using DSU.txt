bool detectCycle(int V, vector<int>adj[])
{
    disjointSet ds(V);
    set<pair<int,int>> edges;
    for(int i=0;i<V;i++){
        for(auto x:adj[i]){
            if(i<x) edges.insert({i,x});
            else edges.insert({x,i});
        }
    }
    for(auto edge:edges){
        
        int u=edge.first;
        int v=edge.second;
        if(ds.findPar(u)==ds.findPar(v)) return true;
        ds.unionBySize(u,v);
    }
    return false;
}
