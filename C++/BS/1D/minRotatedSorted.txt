    int findMin(vector<int>& nums) {
        int l=0,r=nums.size()-1;
        int mini=5001;
        while(l<=r){
            int m = l+(r-l)/2;
            if(nums[l] <= nums[m]){
                mini = min(mini,nums[l]);
                l=m+1;
            }else{
                mini = min(mini,nums[m]);
                r=m-1;
            }
        }
        return mini;
    }
