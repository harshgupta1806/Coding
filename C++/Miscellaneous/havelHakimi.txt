

    bool isValid(vector<int> &ds)
    {
        while (1)
        {
            sort(ds.begin(), ds.end(), greater<int>());
            if (!ds[0])
                return true;
            int s = ds[0];
            ds.erase(ds.begin());
            if (s > ds.size())
                return false;
            for (int i = 0; i < s; i++)
            {
                ds[i]--;
                if (ds[i] < 0)
                    return false;
            }
        }
    }
