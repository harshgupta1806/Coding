328.Given the head of a singly linked list, group all the nodes with odd indices together followed by the nodes with even indices, and return the reordered list.


138. Copy List with Random Pointer
    Node* copyRandomList(Node* head) {
        if(!head){
            return nullptr;
        }
        map<Node*, Node*> mpp;
        Node* temp = head;
        while (temp) {
            mpp[temp] = new Node(temp->val);
            temp = temp->next;
        }
        
        temp = head;
        while (temp) {
            mpp[temp]->next = mpp[temp->next];
            mpp[temp]->random = mpp[temp->random];
            temp = temp->next;
        }
        
        return mpp[head];
    }
