

    string reverseWords (string s)
    {
        stringstream str(s);
        string w,ans="";
        while (getline(str,w,'.')) {
            stack<char> st;
            for(auto c:w) st.push(c);
            string temp="";
            while(!st.empty()) {temp+=st.top();st.pop();}
            ans=ans+temp+'.';
        }
        if(s.back()!='.') ans.pop_back();
        return ans;
    }
