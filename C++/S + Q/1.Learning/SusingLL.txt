class MyStack {
  private:
    StackNode *top;
  public:
    void push(int x) {
      StackNode* newNode = new StackNode(x);
      newNode -> next = top;
      top=newNode;
    }
    int pop() {
      if(!top) return -1;
      int temp = top->data;
      top = top->next;
      return temp;
    }
    MyStack() { top = NULL; }
};
