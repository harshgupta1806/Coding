ListNode *detectCycle(ListNode *head) {
    if(!head || !(head->next)) return NULL;
    ListNode *f=head,*s=head,*t=head;
    do{
        f=f->next;
        if(f) f=f->next;
        s=s->next;
        if(!f){
            return NULL;
        }
    }while(f!=s);
    while(t!=s){
        t=t->next;
        s=s->next;
    }
    return t;
}
