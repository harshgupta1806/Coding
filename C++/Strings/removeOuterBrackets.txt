string removeOuterParentheses(string s) {
    int l=s.length();
    int count=0;
    string res="";
    for(int i=0;i<l;i++){
        if(s[i]=='('){
            if(count) res+=s[i];
            count++;
        }
        if(s[i]==')'){
            if(count!=1)res+=s[i];
            count--;
        }
    }
    return res;
}
