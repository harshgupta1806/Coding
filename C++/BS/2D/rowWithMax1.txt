

    int rowWithMax1s(vector<vector<int> > &arr) {
        int max1=0, ans=-1;
        for(int i=0;i<arr.size();i++){
            int pos = lower_bound(arr[i].begin(),arr[i].end(),1)-arr[i].begin();
            if(arr[i].size() - pos > max1){
                max1=arr[i].size() - pos;
                ans=i;
            }
        }
        return ans;
    }
