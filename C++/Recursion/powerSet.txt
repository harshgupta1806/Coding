    void solve(int i, int n, vector<int> &nums, vector<vector<int>> &all, vector<int> &sbst){
        if(i==n){
            all.push_back(sbst);
            return;
        }
        sbst.push_back(nums[i]);
        solve(i+1,n,nums,all,sbst);
        sbst.pop_back();
        solve(i+1,n,nums,all,sbst);
    }
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<vector<int>> all;
        vector<int> sbst;
        solve(0,nums.size(),nums,all,sbst);
        return all;
    }
